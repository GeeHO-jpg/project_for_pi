== BUILD ==
cd ~/Desktop/project_for_pi/spi_master_comm
./build.sh

== RUN ==
sudo ./spi_test

== RUN ตอน BOOT (systemd service) ==
sudo cp spi_test.service /etc/systemd/system/
sudo systemctl daemon-reload
sudo systemctl enable spi_test.service
sudo systemctl start spi_test.service

-- คำสั่งเพิ่มเติม --
ดูสถานะ:     sudo systemctl status spi_test.service
หยุด:         sudo systemctl stop spi_test.service
ปิด autoboot: sudo systemctl disable spi_test.service
restart:      sudo systemctl restart spi_test.service
