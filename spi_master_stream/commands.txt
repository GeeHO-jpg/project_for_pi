== BUILD ==

chmod +x build.sh

cd ~/Desktop/project_for_pi/spi_master_clean_big_chunk
./build.sh

== RUN ==
sudo ./spi_big_chunk

== RUN ตอน BOOT (systemd service) ==
sudo cp spi_master_clean.service /etc/systemd/system/
sudo systemctl daemon-reload
sudo systemctl enable spi_master_clean
sudo systemctl start spi_master_clean

-- คำสั่งเพิ่มเติม --
ดูสถานะ:     sudo systemctl status spi_master_clean.service
หยุด:         sudo systemctl stop spi_master_clean.service
ปิด autoboot: sudo systemctl disable spi_master_clean.service
restart:      sudo systemctl restart spi_master_clean.service
